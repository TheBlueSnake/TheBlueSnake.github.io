快速傅里叶变换

前置知识：系数表示法和点值表示法

为了唯一确定某一个函数 $f(x)$，我们可以选择使用 $f(x)=\sum_{i=0}^n a_{i}x^{i}$ 的定义，使用 $a_{i}$ 序列定义它，称为系数表示法。也可以通过一系列 $f(x_{i}) = y_{i}$ 的定义，通过离散的序列 $a_{i}, b_{i}$ 唯一确定它，称为点值表示法。

前置知识：虚数与单位根

虚数是对实数域的扩增，分为实部和虚部，记做 $z=a+bi$，这部分内容是高中数学书课内内容，不再赘述。单位根是方程 $x^{n}=1$ 的 $n$ 个解，映射到平面坐标系后，以极角从小到大第 $i$ 个记做 $\omega_{n}^i$。单位根一定在以原点为圆心的单位圆上，通过作图就可以证明所有需要的定理和引理，同样不再赘述。

傅里叶变换

不管是上课过程中还是我看的许多博客中，大家都将傅里叶变换理解为两个多项式相乘的算法，实际上我学习后更倾向于将傅里叶理解为计算出对于 $f(x)$ 函数，给你 $n$ 个单位根 $\omega_{n}^{i}$，请你计算出对应的 $n$ 个函数值 $f(\omega_{n}^{i})$。如果使用虚树的计算直接算的话是 $O(n^2)$ 的复杂度，所以我们需要利用单位根的特殊性质，这就是傅里叶变换。

我们发现，如果我们可以计算出 $f_{1}(\omega_{n}^{2i})$ 与 $f_{2}(\omega_{n}^{2i})$，那么我们同时可以计算出在两个点值。如果采用递归的方式计算答案，那么每一层需要计算 $n$ 个不同值时的函数值，由于坐标系中相对的数可以通过简单的运算整合同时计算出来，所以每 $2$ 个值需要向下递归 $2$ 个不同值，即每向下递归一层数据规模不变，总共会有 $O(\log n)$ 层，因此整体复杂度为 $O(n\log n)$。

逆傅里叶变换

傅里叶变换将系数表示法变换为了点值表示法，因此其逆操作就是将点值表示法转换为系数表示法。

如果我们现在有 $f(x)=a_{0}+a_{1}x+a_{2}x^2+\cdots+a_{n-1}x^{n-1}$ 的系数表示法计算出的 $b_{0}, b_{2}, \dots, b_{n}$。那么如果我们定义 $c_{k}=\sum_{i=0} ^{n-1}b_{i}(\omega_{n}^{-k})^i$，我们可以发现就是将 $\{b_{i}\}$ 作为系数得到 $g(x)$，$\{c_{i}\}$ 就是 $g(x)$ 的一个点值表示法，其取 $c_{i}=f(\omega_{n}^{-i})$。接下来将开始证明 $c_{k}=n\cdot a_{k}$。

代码实现[递归]

从上面的证明我们可以看到，逆傅里叶变换不过是傅里叶变换在计算的时候将所有单位根的指数取负。因此只要在函数中加一个变量，计算的时候便可以使用同一个函数了。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 4000100;
//const int MOD = 1000000007;
//const int MOD = 998244353;
typedef long long ll;
typedef pair<int, int> pii;
typedef long double ldouble;
const double pi = acos(-1.0);
//const ldouble eps = 1e-7;
//inline ll qmod(ll x){return x > MOD ? x - MOD : x;}

struct cplx{
    double a, b;
    cplx(double aa = 0.0, double ab = 0.0){
        a = aa; b = ab;
    }
    cplx operator+ (cplx x){return cplx(a + x.a, b + x.b);}
    cplx operator- (cplx x){return cplx(a - x.a, b - x.b);}
    cplx operator* (cplx x){return cplx(a * x.a - b * x.b, a * x.b + b * x.a);}
};

int n, m, bias, dig;
cplx arr1[MAXN], arr2[MAXN], ans[MAXN];

void fft(int len, cplx *a, int type){
    if(len == 1) return;
    cplx odd[(len >> 1) + 1], eve[(len >> 1) + 1];
    for(int i = 0; i <= len; i += 2){
        odd[i >> 1] = a[i];
        eve[i >> 1] = a[i + 1];
    }
    fft(len >> 1, odd, type);
    fft(len >> 1, eve, type);
    cplx omega = cplx(cos(2.0 * pi / len), type * sin(2.0 * pi / len));
    cplx w = cplx(1.0, 0.0);
    for(int i = 0; i < (len >> 1); i++, w = w * omega){
        a[i] = odd[i] + w * eve[i];
        a[i + (len >> 1)] = odd[i] - w * eve[i];
    }
}

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0; i <= n; i++){
        scanf("%lf", &arr1[i].a);
    }
    for(int i = 0; i <= m; i++){
        scanf("%lf", &arr2[i].a);
    }
    for(bias = 1, dig = 0; bias <= n + m; bias <<= 1, dig++);
    fft(bias, arr1, 1);
    fft(bias, arr2, 1);
    for(int i = 0; i <= bias; i++) ans[i] = arr1[i] * arr2[i];
    fft(bias, ans, -1);
    for(int i = 0; i <= n + m; i++){
        printf("%d ", (int)(ans[i].a / bias + 0.5));
    }
    return 0;
}

蝴蝶迭代

如果我们直接像上面使用递归计算傅里叶变换，我们会超时（奇怪的是我并没有超时，仅仅是最后三个点RE了，可能由于洛谷的栈空间大小或者其他什么问题吧）。所以这时候我们需要非递归型的快速傅里叶变换，那就是蝴蝶迭代。

具体的蝴蝶迭代的证明好像比较麻烦，反正我没有找到一个完整详细的证明方式，所以暂时直接说结论：如果我们仔细观察就可以发现，在计算过程中，不断分割递归分割递归，最终原序列和反转后的序列的下标互为二进制反转，因此不需要将下标不断按照奇偶分割，可以直接通过某种方式得到最终要求的序列，不断向上合并就可以了。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 4000100;
//const int MOD = 1000000007;
//const int MOD = 998244353;
typedef long long ll;
typedef pair<int, int> pii;
typedef long double ldouble;
const double pi = acos(-1.0);
//const ldouble eps = 1e-7;
//inline ll qmod(ll x){return x > MOD ? x - MOD : x;}

struct cplx{
    double a, b;
    cplx(double aa = 0.0, double ab = 0.0){
        a = aa; b = ab;
    }
    cplx operator+ (cplx x){return cplx(a + x.a, b + x.b);}
    cplx operator- (cplx x){return cplx(a - x.a, b - x.b);}
    cplx operator* (cplx x){return cplx(a * x.a - b * x.b, a * x.b + b * x.a);}
};

int n, m, bias, dig, pos[MAXN];
cplx arr1[MAXN], arr2[MAXN], ans[MAXN];

void fft(cplx *a, int type){
    for(int i = 0; i <= bias; i++){
        if(i < pos[i]){
            swap(a[i], a[pos[i]]);
        }
    }
    for(int len = 1; len < bias; len <<= 1){
        cplx omega = cplx(cos(pi / len), type * sin(pi / len));
        for(int r = len << 1, j = 0; j < bias; j += r){
            cplx w = cplx(1.0, 0.0);
            for(int k = 0; k < len; k++, w = w * omega){
                cplx lft = a[j + k], rht = w * a[j + len + k];
                a[j + k] = lft + rht;
                a[j + len + k] = lft - rht;
            }
        }
    }
}

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0; i <= n; i++){
        scanf("%lf", &arr1[i].a);
    }
    for(int i = 0; i <= m; i++){
        scanf("%lf", &arr2[i].a);
    }
    for(bias = 1, dig = 0; bias <= n + m; bias <<= 1, dig++);
    for(int i = 0; i <= bias; i++){
        pos[i] = (pos[i >> 1] >> 1) | ((i & 1) << (dig - 1));
    }
    fft(arr1, 1);
    fft(arr2, 1);
    for(int i = 0; i <= bias; i++){
        ans[i] = arr1[i] * arr2[i];
    }
    fft(ans, -1);
    for(int i = 0; i <= n + m; i++){
        printf("%d ", (int)(ans[i].a / bias + 0.5));
    }
    return 0;
}

注意，因为运算的时候我们将 $n$ 上取整到 $2$ 的幂次，所以最保险的情况下，我们应该开 $4n$ 的数组空间，当然如果内存比较少的话，只需要开到距离 $n+m$ 最近且大于它的 $2$ 的整数次幂两倍即可。

快速数论变换

快速傅里叶变换能够在 $O(n\log n)$ 的时间复杂度内将多项式在点值表示法与系数表示法之间转换。但是如果我们处理的是整系数多项式，题目要求我们对一个数取模，我们就不可以通过虚树计算了。同时即使不需要取模，浮点数在计算大数的时候也会导致精度较低，答案会有误差。因此我们需要快速数论变换来计算。

前置知识：原根

如果 $g$ 是 $p$ 的原根，则存在 $g^{i} \not\equiv g^{j} \mod p\quad(i\neq j, 1\leq i,j\leq p-1)$。换句话说，原根 $g$ 能够使 $\left\{g,g^2,\dots,g^{p}\right\}$ 集合内中的元素与 $\left\{1,2,\dots,p\right\}$ 集合内的元素有一一对应关系。

同时，如果 $p$ 可以表示成 $p=a\times 2^k+1$ 的形式，如果我们记 $g_{n}=g^{\frac{p-1}{n}}$，存在以下结论：

这些结论与 $\textsf{FFT}$ 中使用的单位根完全相同，因此可以用来替换单位根。计算过程万完全相同，只是不需要涉及浮点数和虚数了，因此精度会提高而且可以在 $p$ 是质数的时候取模了。

另外，常用模数 $998244353$ 的原根是 $3$。

代码实现

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 4000100;
const int MOD = 998244353;
//const int MOD = 1000000007;
typedef long long ll;
typedef pair<int, int> pii;
typedef long double ldouble;
const ll g = 3;
const ll invg = 332748118;
//const ldouble eps = 1e-7;
//inline ll qmod(ll x){return x > MOD ? x - MOD : x;}

int n, m, bias, dig, pos[MAXN];
ll arr1[MAXN], arr2[MAXN], ans[MAXN];

ll qpow(ll x, int pwr){
    ll res = 1;
    while(pwr){
        if(pwr & 1) res = res * x % MOD;
        x = x * x % MOD;
        pwr >>= 1;
    }
    return res;
}

void ntt(ll *a, int type){
    for(int i = 0; i <= bias; i++){
        if(i < pos[i]){
            swap(a[i], a[pos[i]]);
        }
    }
    for(int len = 1; len < bias; len <<= 1){
        ll omega = qpow(type == 1 ? g : invg, (MOD - 1) / (len << 1));
        for(int j = 0; j < bias; j += len << 1){
            ll gn = 1;
            for(int k = 0; k < len; k++, gn = gn * omega % MOD){
                ll lft = a[j + k], rht = gn * a[j + k + len] % MOD;
                a[j + k] = (lft + rht) % MOD;
                a[j + k + len] = (lft - rht + MOD) % MOD;
            }
        }
    }
}

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0; i <= n; i++){
        scanf("%lld", &arr1[i]);
    }
    for(int i = 0; i <= m; i++){
        scanf("%lld", &arr2[i]);
    }
    for(bias = 1, dig = 0; bias <= n + m; bias <<= 1, dig++);
    for(int i = 0; i <= bias; i++){
        pos[i] = (pos[i >> 1] >> 1) | ((i & 1) << (dig - 1));
    }
    ntt(arr1, 1);
    ntt(arr2, 1);
    for(int i = 0; i <= bias; i++){
        ans[i] = arr1[i] * arr2[i] % MOD;
    }
    ntt(ans, -1);
    ll inv = qpow(bias, MOD - 2);
    for(int i = 0; i <= n + m; i++){
        printf("%lld ", ans[i] * inv % MOD);
    }
    return 0;
}
