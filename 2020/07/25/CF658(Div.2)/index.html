A. Common Subsequence

题目说最短公共子序列。我们可以发现，对于任意公共子序列，一定存在长度为$1$，也就是单独一个字母的公共子序列。所以题目就变成了看存不存在相同数字。使用类似归并排序合并的方式，两个序列排序后，分别从左往右两个指针扫一下就可以了。复杂度$O(n+m)$

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1100;

int t;
int n, m, a[MAXN], b[MAXN];

bool cmp(const int &x, const int &y){return x < y;}

int main(){
    scanf("%d", &t); while(t--){
        scanf("%d %d", &n, &m);
        for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
        for(int i = 1; i <= m; i++) scanf("%d", &b[i]);
        bool flag = false;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                if(a[i] == b[j]){
                     flag = true;
                     printf("YES\n");
                     printf("%d %d\n", 1, a[i]);
                     break;
                }
            }
            if(flag) break;
        }
        if(!flag) printf("NO\n");
    }
    return 0;
}

B. Sequential Nim

又是一道博弈论的题目，对于一个$a_i=1$的石子堆，先后手必定发生交换，对于一个$a_i\neq 1$的石子堆，先手可以决定先后手是否发生交换。所以对于一连串$a_i\neq 1$的石子堆，先手一定可以通过取$a_i-1$个，随后剩下一个让对方取来保证自己先手位置，于是便等价于只有一堆$a_i\neq1$的石子堆。由于题目可知必定存在必胜或必败策略，我们可以知道，影响新的答案的石子堆只会是开头第一串$a_i=1$的石子和开头第一串$a_i\neq 1$的石子。每次会交换一次先后手，先摸到$a_i\neq1$的人就是必胜的。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100100;

int T;
int n, arr[MAXN];

int main(){
    scanf("%d", &T); while(T--){
        scanf("%d", &n);
        for(int i = 1; i <= n; i++) scanf("%d", &arr[i]);
        bool flag = false;
        for(int i = 1; i <= n; i++){
            if(arr[i] == 1) flag ^= 1;
            else{
                flag ^= 1;
                break;
            }
        }
        printf("%s\n", flag ? "First" : "Second");
    }
    return 0;
}

C2. Prefix Flip (Hard Version)

由于没有想到$2n<k\leq 3n$的做法，在这里直接说$k\leq 2n$的做法了。
从尾巴往前面扫，如果当前位置$a_i=b_i$那么就不用去改变了，继续往前扫，否则的话一定可以通过以下操作，在两步以内，将当前位置变为$a_i=b_i$，操作如下：
只用将$1\to i$全部翻转一下，就可以将$a_i\oplus 1$转移到第$i$位，所以只用判断是否需要改变第一位的值，如果$a_1\neq b_i$直接进行$reverse(i)$，否则进行$reverse(1)$然后进行$reverse(i)$。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100100;

int T;
int n, firstpos;
char arr[MAXN], b[MAXN];
int ans, rev; queue<int> que, str;

int chk(int val){
    if(!rev) return firstpos + val - 1;
    else return firstpos - val + 1;
}

int main(){
//    freopen("C.in", "r", stdin);
//    freopen("C.out", "w", stdout);
    scanf("%d", &T); int tmps = T; while(T--){
        scanf("%d", &n);
        scanf("%s", arr + 1);
        scanf("%s", b + 1);
        firstpos = 1;
        ans = rev = 0;
        while(!que.empty()) que.pop();
        for(int i = n; i >= 1; i--){
            if(((arr[chk(i)] - '0') ^ (rev)) != (b[i] - '0')){
//                if(tmps == 1) printf(":inside %d\n", i);
//                printf("%d %d %d\n", i, ans, rev);
                if(((arr[chk(1)] - '0') ^ (rev)) == (b[i] - '0')){
                    que.push(1);
                    ans++;
                }
                firstpos = chk(i);
                que.push(i);
                ans++; rev ^= 1;
            }
        }
        printf("%d ", ans);
        while(!que.empty()){
            printf("%d ", que.front());
            que.pop();
        }
        printf("\n");
    }
    return 0;
}

D. Unmerge

我们可以发现，原序列可以化为多段子序列拼在一起的结果，对于每一段单调下降子序列，显然必定是在单独一个序列$a$或$b$内的，如果不是的话，那么最大的数会最后放出来。所以我们很容易想到一种方法，就是将序列中最大的数找出来，然后以这个数做起点，向后所有的数都拉出来，这一段数一定是放在一个子序列内的，不然当前数会比后面的数更晚合并。不妨记至此分割为$blk$块，我们只用求这$blk$块可不可以拼成$n$。构造便是将这些序列放在$a$中，然后将剩下的部分依次放到另一个序列$b$中即可。
考试的时候脑残，在$\sum n\leq 2000$的时候，我们可以证明$O(n^2)$复杂度是可以通过的。不妨设$T$次分别为$n_1,n_2,\dots,n_T$。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 4100;

int T;
int n, arr[MAXN], pos[MAXN], exist[MAXN];
int blk[MAXN], f[MAXN][MAXN];

int main(){
    scanf("%d", &T); while(T--){
        scanf("%d", &n); n <<= 1;
        for(int i = 1; i <= n; i++){
            scanf("%d", &arr[i]);
            pos[arr[i]] = i;
        }
        for(int i = 1; i <= n; i++) exist[i] = -1;
        blk[0] = 0;
        for(int prev = n + 1, i = n; i >= 1; i--){
            if(exist[i] == -1){
                exist[i] = 1;
                for(int j = pos[i]; j <= prev - 1; j++)
                    exist[arr[j]] = 1;
                blk[++blk[0]] = prev - pos[i];
                prev = pos[i];
            }
        }
//        printf(":");
//        for(int i = 1; i <= blk[0]; i++)
//            printf("%d ", blk[i]);
//        printf("\n");
        f[0][0] = 1;
        for(int i = 1; i <= blk[0]; i++){
            for(int j = 0; j <= (n >> 1); j++){
                f[i][j] = f[i - 1][j];
                if(j - blk[i] >= 0) f[i][j] |= f[i - 1][j - blk[i]];
            }
        }
        printf("%s\n", f[blk[0]][n >> 1] ? "YES" : "NO");
    }
    return 0;
}
