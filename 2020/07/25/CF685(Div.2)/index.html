A.Subtract or Divide

给你一个数，每次可以将它除以不为自身的任意一个约数，或者可以减一，问最少多少次操作可以变成1。其实是很水的，但是最开始做错了，两次……（害怕qwq。正解为特判 $n=1$ 答案是 $0$， $n=2$ 答案是 $1$， $n=3$ 答案是 $4$。其他情况下，如果 $n$ 是偶数，最优操作一定是将它除以 $\frac n 2$ 变为 $2$，再一次操作变为 $1$。如果 $n$ 是偶数，最优解一定是将它减一，除以 $\frac{n-1}2$，并再减一得到 $1$。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
//const int MAXN = ;
typedef long long ll;
typedef pair<int, int> pii;

int n;

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d", &n);
    if(n == 1){
        printf("0\n");
        continue;
    }else if(n == 2){
        printf("1\n");
        continue;
    }else if(n == 3){
        printf("2\n");
        continue;
    }else if(n % 2 == 0){
        printf("2\n");
        continue;
    }else if(n % 2 == 1){
        printf("3\n");
        continue;
    }
}
    return 0;
}

B. Non-Substring Subsequence

最开始读错题，以为优秀的联通块是指最大联通长度不超过 $2$，浪费了两次提交。

其实只要满足 $a_{l_i}$ 是第一个 $0$ 或者第一个 $1$ 或者 $a_{r_i}$ 是最后一个 $0$ 或者最后一个 $1$ 不同时满足，就一定可以满足条件，只要把首尾的那个数换成更前面或者更后面的 $0$ 或 $1$ 即可。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 110;
typedef long long ll;
typedef pair<int, int> pii;

int n, q;
char str[MAXN];

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d %d", &n, &q);
    scanf("%s", str + 1);
    int fst1 = -1, fst0 = -1, lst1 = -1, lst0 = -1;
    for(int i = 1; i <= n; i++){
        if(str[i] == '1'){
            fst1 = i;
            break;
        }
    }
    for(int i = 1; i <= n; i++){
        if(str[i] == '0'){
            fst0 = i;
            break;
        }
    }
    for(int i = n; i >= 1; i--){
        if(str[i] == '1'){
            lst1 = i;
            break;
        }
    }
    for(int i = n; i >= 1; i--){
        if(str[i] == '0'){
            lst0 = i;
            break;
        }
    }
//    printf(":%d %d %d %d\n", fst0, fst1, lst0, lst1);
    for(int li, ri, i = 1; i <= q; i++){
        scanf("%d %d", &li, &ri);
        if(ri - li + 1 < 2){
            printf("NO\n");
            continue;
        }else if((li == fst1 || li == fst0) && (ri == lst1 || ri == lst0)){
            printf("NO\n");
            continue;
        }else{
            printf("YES\n");
            continue;
        }
    }
}
    return 0;
}

C. String Equality

最开始以为最不可做，结果是唯一一个一次提交就过了的题目。由于每次操作必须是 $k$ 次变换，我们不妨先将 $a$ 串用 swap 操作变为有序，从连续的 $a$ 开始操作，每次变 $k$ 个，如果每次都可以正好留下 $b$ 串中该字符个数个字符，就接着往下做，否则就不可以做到。全部变换完成后，再用 swap 操作换成 $b$ 顺序即可。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 1000100;
typedef long long ll;
typedef pair<int, int> pii;

int n, k;
char str[MAXN];
int l1[30], l2[30];

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d %d", &n, &k);
    memset(l1, 0, sizeof(l1));
    memset(l2, 0, sizeof(l2));
    scanf("%s", str + 1);
    for(int i = 1; i <= n; i++){
        l1[str[i] - 'a' + 1]++;
    }
    scanf("%s", str + 1);
    for(int i = 1; i <= n; i++){
        l2[str[i] - 'a' + 1]++;
    }
    bool flag = false;
    for(int i = 1; i <= 26; i++){
//        printf(":%c: %d %d\n", 'a' + i - 1, l1[i], l2[i]);
        if(l1[i] - l2[i] >= 0 && (l1[i] - l2[i]) % k == 0){
            l1[i + 1] += l1[i] - l2[i];
        }else{
            printf("No\n");
            flag = true;
            break;
        }
    }
    if(!flag){
        printf("Yes\n");
        continue;
    }
}
    return 0;
}

D. Circle Game

最开始看到这道题脑子还是蒙的。事实证明，我用一个口胡的奇怪做法过了这道题，然后貌似还没有FST。我的想法就是先把圆按照比例缩小。将半径 $d$ 除以每次走的长度 $d$，此时就变成了一个半径为 $\frac d k$ 的圆，每次走一格，问谁会赢。（这两步之间应该是等价的吧？）此时我们只要找出可以走的整数格子中距离原点曼哈顿距离最大的位置。所以先找中间的 $(p,p)$ 这个坐标轴角平分线上的对称的点，然后看 $(p+1,p)$ 或者 $(p,p+1)$ 是不是也在圆内，是的话就说明最远的点在旁边，不然的话就是这个点。然后只要判断奇偶性就可以得出谁获胜了，因为不存在会被压制的情况，所以直接输出即可。

尝试过 long long，尝试过把除法移过去变为乘法。最终还是向 double 屈服了……

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
//const int MAXN = ;
typedef long long ll;
typedef pair<int, int> pii;

int di, k;

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d %d", &di, &k);
    double dis = floor(1.0 * di / sqrt(2) / k);
    if(1.0 * dis * dis + 1.0 * (dis + 1) * (dis + 1) <= 1.0 * di * di / k / k){
        printf("Ashish\n");
    }else{
        printf("Utkarsh\n");
    }
}
    return 0;
}

E1. Bitwise Queries (Easy Version)

暂时只做出来了Easy Version。不过看了官方题解，口胡出了Hard Version。

其实考场上看到是一道交互题就不想做了，但是看了看F口胡一个算法，但是细节貌似时间不够了就回来看E。

先说我考场上的想法，我最开始以为三种操作是给你从 $a_i$ 一直异或异或异或到 $a_j$ 的区间查询，然后就写了一个询问 $a_1 \oplus a_2$，$a_2 \oplus a_3$，$a_1\oplus a_2\oplus a_3$，然后就算出来$a_1$，$a_2$，$a_3$美滋滋，直接一路异或到 $a_n$ 就可以求出全部答案了。一交，三号点错了（谢天谢地，不然就FST了）。然后心态崩了，反复调试，然后才又读了一遍题，发现只是询问 $a_i \oplus a_j$。

于是想到了另一种做法，我考虑到所有位运算都不存在方向性（就是指 $f(x,y)=f(y,x)$），所以我要想办法把它换一种操作去做，于是就想到了加减法。$a+b=(a \oplus b)+((a \and b) << 1)$。原理就是两者都是 $1$ 所以要进位。于是我们用四次操作得到了 $a_1+a_2$，$a_2+a_3$。但是有操作次数限制，从 $a_4$ 开始每次需要一个异或询问来得到结果。（其实中间卡在思考题目保证的 $n$ 是 $2$ 指数次幂有什么用）。所以只留下 $5$ 次操作给前三个数，这时候我们可以省下来一次操作，因为 $a_1\oplus a_3=(a_1\oplus a_2)\oplus(a_2\oplus a_3)$，所以直接通过前两次计算得到了 $a_1\oplus a_3$，只需要询问 $a_1\and a_3$ 就可以计算出 $a_1+a_3$，$a_1+a_2+a_3=\frac{(a_1+a_2)+(a_2+a_3)+(a_3+a_1)}2$，每个相减即可得到 $a_1,a_2,a_3$。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 100100;
typedef long long ll;
typedef pair<int, int> pii;

int n;
int arr[MAXN];

int xxor(int l, int r){
    int ret;
    printf("XOR %d %d\n", l, r);
    fflush(stdout);
    scanf("%d", &ret);
    return ret;
}

int aand(int l, int r){
    int ret;
    printf("AND %d %d\n", l, r);
    fflush(stdout);
    scanf("%d", &ret);
    return ret;
}

int oor(int l, int r){
    int ret;
    printf("OR %d %d\n", l, r);
    fflush(stdout);
    scanf("%d", &ret);
    return ret;
}

int main(){
    scanf("%d", &n);
    int xxor12 = xxor(1, 2), aand12 = aand(1, 2);
    int xxor23 = xxor(2, 3), aand23 = aand(2, 3);
    int sum12 = xxor12 + (aand12 << 1);
    int sum23 = xxor23 + (aand23 << 1);
    int xxor13 = xxor12 ^ xxor23;
    int aand13 = aand(1, 3);
    int sum13 = xxor13 + (aand13 << 1);
    int sum123 = (sum12 + sum23 + sum13) / 2;
    arr[1] = sum123 - sum23;
    arr[2] = sum123 - sum13;
    arr[3] = sum123 - sum12;
    for(int i = 4; i <= n; i++){
        arr[i] = arr[i - 1] ^ xxor(i - 1, i);
    }
    printf("! ");
    for(int i = 1; i <= n; i++){
        printf("%d ", arr[i]);
    }
    fflush(stdout);
    return 0;
}

E1. Bitwise Queries (Hard Version)

其实就是上面的做法，只不过因为值域在 $[0,n-1]$，所以要么所有数都不相同，是全排列，要么是存在相同的数。通过一遍相邻异或并计算出所有数之间的异或，此时操作 $n-1$ 次，如果有两者相同，那么这两个数相同，可以比上面省下来一次 $a_i\and a_j$，总共 $n+1$ 次询问。如果没有数相同，就利用 $n$ 和之间关系直接推导 $n$ 个数（异或之间确定），总计 $n-1$ 次询问。
