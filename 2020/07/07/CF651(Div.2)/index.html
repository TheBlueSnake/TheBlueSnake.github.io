A. Maximum GCD

水，很明显满足$1\leq a< n\leq n$且$\gcd(a,b)$最大的数，就是$\gcd(2\times\left\lfloor\dfrac n 2\right\rfloor,\left\lfloor\dfrac n 2\right\rfloor)=\left\lfloor\dfrac n 2\right\rfloor$直接输出即可。

#include<bits/stdc++.h>
using namespace std;

int T;
int n;

int main(){
    scanf("%d", &T); while(T--){
        scanf("%d", &n);
        printf("%d\n", n >> 1);
    }
    return 0;
}

B. GCD Compression

考试时脑抽了，这么水的一道题竟然没有A。
首先对于任意的$\left\{a_i\right\}$，一定存在一种分配方式，使$\gcd(b_1,b_2,\dots,b_n)=2>1$，证明：
不妨考虑将$\left\{a_i\right\}$中所有的数分为两个集合$A=\left\{x|x\in S,x=2k+1,k\in Z\right\}$和$B=\left\{x|x\in S,x=2k,k\in Z\right\}$，那么如果$\left|A\right|$和$\left|B\right|$均为偶数，分别从中两个两个取，$\gcd(b_1,b_2,\dots,b_n)=2$，满足条件；如果$\left|A\right|$和$\left|B\right|$均为奇数，分别丢弃一个，然后两个两个取，$\gcd(b_1,b_2,\dots,b_n)=2$，满足条件。
所以对于任意$\left\{a_i\right\}$，一定存在一种分配方式，使$\gcd(b_1,b_2,\dots,b_n)=2>1$。
有了这个结论，我们直接按照证明过程去做就行了：

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 2100;

int T;
int n, arr[MAXN], odd[MAXN], even[MAXN];

int main(){
    scanf("%d", &T); while(T--){
        scanf("%d", &n); n <<= 1;
        odd[0] = even[0] = 0;
        for(int i = 1; i <= n; i++){
            scanf("%d", &arr[i]);
            if(arr[i] % 2 == 1) odd[++odd[0]] = i;
            else even[++even[0]] = i;
        }
//        printf(":%d %d\n", odd[0], even[0]);
        if(odd[0] % 2 == 0 && even[0] % 2 == 0){
            if(odd[0] != 0){
                for(int i = 1; i + 1 <= odd[0] - 2; i += 2) printf("%d %d\n", odd[i], odd[i + 1]);
                for(int i = 1; i + 1 <= even[0]; i += 2) printf("%d %d\n", even[i], even[i + 1]);
            }else{
                for(int i = 1; i + 1 <= odd[0]; i += 2) printf("%d %d\n", odd[i], odd[i + 1]);
                for(int i = 1; i + 1 <= even[0] - 2; i += 2) printf("%d %d\n", even[i], even[i + 1]);
            }
        }else{
            for(int i = 1; i + 1 <= odd[0]; i += 2) printf("%d %d\n", odd[i], odd[i + 1]);
            for(int i = 1; i + 1 <= even[0]; i += 2) printf("%d %d\n", even[i], even[i + 1]);
        }
    }
    return 0;
}

C. Number Game

很烦的博弈论题目，首先将$n$按照唯一分解定理分解：

我们不需要考虑这么多，只用考虑

  1. 如果$k=0$，$p=1$，明显先手无法操作，后手必胜。
  2. 如果$k=0$，$p\neq1$，先手除以$n$，得到$1$，后手无法操作，先手必胜。
  3. 如果$k=1$，$p=1$，先手减去$1$，得到$1$，后手无法操作，先手必胜。
  4. 如果$k=1$,$p$是一个奇质数，先手如果减去$1$，得到一个奇数，先手必输；先手如果除以$p$得到$2$，先手必输。
  5. 如果$k=1$，$p$是多个质数的乘积，先手将其除以$p$的第二大约数，变为第$4$钟情况，先手必胜。
  6. 如果$k>1$，$p=1$，先手只能减去$1$，得到一个奇数，先手必输。
  7. 如果$k>1$，$p\neq 1$，先手除以$p$，转变为情况$6$，先手必胜。

直接分类判断即可

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

ll T;
ll n;

int main(){
    scanf("%lld", &T); while(T--){
        scanf("%lld", &n); ll ans = 0;
        ll remainer = n, two_pwr = 1, pwr = 0;
        if(remainer == 1){printf("FastestFinger\n"); continue;}
        else if(remainer % 2 == 1){printf("Ashishgup\n"); continue;}
        while(remainer % 2 == 0) two_pwr <<= 1, remainer >>= 1;
        for(int i = 3; i * i <= remainer; i++)
            while(remainer % i == 0) pwr++, remainer /= i;
        if(remainer != 1) pwr++;
        if(pwr == 0 && two_pwr == 2){printf("Ashishgup\n"); continue;}
        else if(pwr == 0 && two_pwr != 2){printf("FastestFinger\n"); continue;}
        else if(pwr == 1 && two_pwr == 2){printf("FastestFinger\n"); continue;}
        else if(pwr == 1 && two_pwr != 2){printf("Ashishgup\n"); continue;}
        else if(pwr >= 2 && two_pwr == 2){printf("Ashishgup\n"); continue;}
        else if(pwr >= 2 && two_pwr != 2){printf("Ashishgup\n"); continue;}
    }
    return 0;
}

D. Odd-Even Subsequence

读完题很明显二分答案，$O(n)$确定为复杂度的判断，具体操作是确定当前端点，往右扩张，能加入的就加入，最后计算答案是不是满足条件。需要注意的是一定要判断两次，一次判断在奇数位，一次判断在偶数位。同时还要注意判断过程中起始点和终止点的边界诶条件，偶数位需要预留出第一个位置，同时在$k$是奇数的时候预留出最后一个位置；奇数位在$k$是偶数的时候也需要预留出最后一个位置。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200100;
const int INF = 0x3f3f3f3f;

int n, k, arr[MAXN];

bool judge1(int x){ //1, 3, 5, 7, ...
    int demand = ((k + 1) >> 1);
    for(int i = 1; i <= n - (k % 2 == 0); i++){
        if(arr[i] <= x) demand--, i++;
    }
    return demand <= 0;
}
bool judge2(int x){
    int demand = (k >> 1);
    for(int i = 2; i <= n - (k % 2 == 1); i++){
        if(arr[i] <= x) demand--, i++;
    }
    return demand <= 0;
}

int main(){
    scanf("%d %d", &n, &k);
    for(int i = 1; i <= n; i++) scanf("%d", &arr[i]);
    int lft = 1, rht = INF, mid, ans = INF + 1;
    while(lft <= rht){
        mid = (lft + rht) >> 1;
        if(judge1(mid) || judge2(mid)) ans = mid, rht = mid - 1;
        else lft = mid + 1;
    }
    printf("%d\n", ans);
    return 0;
}

E. Binary Subsequence Rotation

题目中$S$串要转换到$T$串。如果当前位置$S_i=T_i$很明显忽略掉这一位一定对于答案没有影响。除此之外，我们可以将剩余的位置压缩为$S’$和$T’$两条$01$串，我们每次操作最优就是将$1000\cdots0001000\cdots0001000\cdots$串右移一位，最终得到的就是其中右移位置中最长的$01$子串的长度，因此我们只需要将$S’$串的$0$记做$1$，将$T’$串中的$0$记做$-1$，每次遇到一个就相当于两者之间差距拉开或缩小了$1$，最终答案就是差距可能的最大值减去差距可能的最小值。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1000100;
const int INF = 0x3f3f3f3f;

int n;
char str1[MAXN], str2[MAXN];
int cnt, maximum, minimum;

int main(){
    scanf("%d", &n);
    scanf("%s", str1 + 1);
    scanf("%s", str2 + 1);
    int cnt1 = 0, cnt2 = 0;
    for(int i = 1; i <= n; i++){
        if(str1[i] == '1') cnt1++;
        if(str2[i] == '1') cnt2++;
    }
//    printf(":%d %d\n", cnt1, cnt2);
    if(cnt1 != cnt2){
        printf("-1\n");
        return 0;
    }
    cnt = 0; maximum = 0; minimum = INF;
    for(int i = 1; i <= n; i++){
        cnt += str1[i] == '1' ? 0 : 1;
        cnt -= str2[i] == '0' ? 1 : 0;
//        printf(":%d %d %d\n", maximum, minimum, cnt);
        maximum = max(maximum, cnt);
        minimum = min(minimum, cnt);
    }
    printf("%d\n", abs(maximum - minimum));
    return 0;
}
