A. FashionabLee

很水的一道数学题，一个正$n$边形，那么每条边所对的中心角为$\dfrac{360} n$，如果要满足两条边分别和$OX$轴和$OY$轴平行，就要满足存在$x\in Z^+$，使$\dfrac{360}n=\dfrac{90}{x}$，即$\left.\dfrac{360}n\middle|90\right.$，浮点数不好参与运算，因此改为$\left.360\middle|\dfrac{90}n\right.$，直接判断即可。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int t;
ll n;

int main(){
    scanf("%d", &t); while(t--){
        scanf("%lld", &n);
        if((90 * n) % 360 == 0) printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}

B. AccurateLee

要缩减$1$和$0$，我们可以很容易得出一种最优的缩减方式：
开头的$0$我们无法操作其减少，末尾的$1$我们课无法操作其减少，除此之外，对于中间的$01$序列，将第一段$111\cdots111000\cdots000$变换为$1$，其余的变换为$0$，最终得到$1000\cdots000$序列可以变为$0$。显然：该种变换方式是字典序最小的。直接按照其计算即可，需要注意特判$000\cdots000111\cdots111$的特殊情况，中间不需要加0。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100100;

int t;
int n; char str[MAXN];

int main(){
    scanf("%d", &t); while(t--){
        scanf("%d", &n);
        scanf("%s", str + 1);
        int xpos = n, ypos = 1;
        for(int i = 1; i <= n; i++){
            if(str[i] == '1'){
                xpos = i - 1;
                break;
            }
        }
        for(int i = n; i >= 1; i--){
            if(str[i] == '0'){
                ypos = i + 1;
                break;
            }
        }
//        printf(":%d %d\n", xpos, ypos);
        for(int i = 1; i <= xpos; i++) printf("0");
        if(ypos != xpos + 1) printf("0");
        for(int i = 1; i <= n - ypos + 1; i++) printf("1");
        printf("\n");
    }
    return 0;
}

C. RationalLee

我们要最大化每个伙伴的开心度之和，因此要最大化选出的最大值和最小值之和。显然先将礼物按照$a_i$从小到大排序。那么直觉告诉我们不妨先将最大的$k$个礼物分别分给$k$个人，得到的最大值一定是最大的。同时将拿礼物多的人先拿最小的$w_i$个礼物，以此尽量最大化后面拿礼物的人的最小值。显然经过这种分配最终得到的幸福度之和最大。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200100;
const int MAXM = 200100;
typedef long long ll;

int t;
int n, k;
ll arr[MAXN], cnt[MAXM], ans = 0;

bool cmp(const ll &x, const ll &y){return x < y;}

int main(){
    scanf("%d", &t); while(t--){
        scanf("%d %d", &n, &k);
        for(int i = 1; i <= n; i++) scanf("%lld", &arr[i]);
        for(int i = 1; i <= k; i++){
            scanf("%lld", &cnt[i]);
            cnt[i]--;
        }
        sort(arr + 1, arr + n + 1, cmp);
        sort(cnt + 1, cnt + k + 1, cmp);
        ans = 0;
        for(int i = 1; i <= k; i++){
            ans += arr[n - i + 1];
            if(cnt[i] == 0) ans += arr[n - i + 1];
        }
        for(int sum = 1, i = k; i >= 1; i--){
//            printf(":%d %d %d\n", cnt[i], sum, ans);
            if(cnt[i] == 0) continue;
            ans += arr[sum];
            sum += cnt[i];
        }
        printf("%lld\n", ans);
    }
    return 0;
}

我们将五级RDB与四级RDB和三级RDB比较一下：

图中是五级RDB，红线圈出部分为四级RDB，蓝线圈出部分为三级RDB。推广一下可以得到$n$级RDB由一个$n-1$级RDB和两个$n-2$级RDB组合而成。
题目要我们求的是树上可以涂色的爪子的数量，那么可以轻易得到一个递推式$f[i]=f[i-1]+2\times f[i-2]$，然而我们使用这个递推式递推的时候却会发现，只有前几个是满足的，大样例答案错误。
经过思考可以很容易推出错误原因，连接几个小RDB的端点$1$可以和几个小RDB组成一个可行的爪子，但是这不一定每次都要加上$4$有时候却不能组成一个爪子，因此我们引入新的$head$数组用来记录最优解时是不是一定要将顶端点涂色，可以得到递推式：

可以$O(n)$预处理后$O(t)$输出答案。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 2000100;
const int MOD = 1e9 + 7;
typedef long long ll;

ll t;
ll n, f[MAXN], head[MAXN];

int main(){
    memset(f, 0, sizeof(f));
    memset(head, 0, sizeof(head));
    for(ll i = 3; i <= 2000000; i++){
        f[i] = (1ll * f[i - 1] + 1ll * f[i - 2] * 2) % MOD;
        if(head[i - 2] == 0 && head[i - 1] == 0){
            f[i] = (f[i] + 4) % MOD;
            head[i] = 1;
        }
    }
    scanf("%lld", &t);
    for(ll inpt, i = 1; i <= t; i++){
        scanf("%lld", &inpt);
        printf("%lld\n", f[inpt]);
    }
    return 0;
}

E. DeadLee

Lee的好朋友（确信）
题目本身其实也不难，可以贪心，假设有一种食物，喜欢这种食物的人的数量甚至比食物本身有的数量还要少，我们可以得到结论：喜欢这种食物的朋友都不会生气到吃掉Lee，因此贪心地将喜欢这种食物的朋友全部都放到尽量后面去，这样他们能不用消耗其他的食物就尽量不消耗其他食物。所以我们可以把喜欢该食物的人喜欢的另一种食物需求量减少$1$，因为就算没有剩下一种食物了，他还是可以吃这种食物（稍微有那么点像拓扑排序），于是借鉴拓扑排序思想一遍一遍滚，如果到最后还是有朋友吃不上饭，那么就真的是吃不上饭了。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100100;
const int MAXM = 200100;

int n, m;
pair<int, int> fri[MAXM];
int w[MAXN];
vector<int> loved[MAXN];
queue<int> que;
int vis[MAXN], exist[MAXM];
vector<int> ans;

inline void remove(int pos){
    if(++w[pos] >= 0 && !vis[pos]){
        que.push(pos);
        vis[pos] = true;
    }
}

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 1; i <= n; i++) scanf("%d", &w[i]);
    for(int i = 1; i <= m; i++){
        scanf("%d %d", &fri[i].first, &fri[i].second);
        w[fri[i].first]--;
        w[fri[i].second]--;
        loved[fri[i].first].push_back(i);
        loved[fri[i].second].push_back(i);
    }
    memset(vis, 0, sizeof(vis));
    memset(exist, 0, sizeof(exist));
    for(int i = 1; i <= n; i++) if(w[i] >= 0) que.push(i), vis[i] = true;
    while(!que.empty()){
        int tp = que.front(); que.pop();
//        printf(":%d\n", tp);
        int length = loved[tp].size();
        for(int i = 0; i < length; i++){
            remove(fri[loved[tp][i]].first);
            remove(fri[loved[tp][i]].second);
            if(!exist[loved[tp][i]]){
                ans.push_back(loved[tp][i]);
//                printf(":push%d\n", loved[tp][i]);
                exist[loved[tp][i]] = true;
            }
        }
    }
    int length = ans.size();
    if(length < m) printf("DEAD\n");
    else{
        printf("ALIVE\n");
        for(int i = length - 1; i >= 0; i--)
            printf("%d ", ans[i]);
        printf("\n");
    }
    return 0;
}
