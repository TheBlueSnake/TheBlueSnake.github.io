自动机

什么是自动机？

自动机一般是确定有限状态自动机DFA（Definite Finite Automata）的数学模型。FSM是根据输入，可以依据转移函数进行状态间的转移的一种机器。信息中常见的自动机有：

  • AC自动机
  • 后缀自动机（广义后缀自动机）
  • 子序列自动机
  • 回文自动机
  • 以及其他不常见的自动机模型

自动机有什么用？

自动机主要运用在字符串的处理上，通常将字符串的某个子串作为自动机的一种状态，其中自动机可以根据需要利用转移函数通过一种状态转移为另一种状态。此时的自动机一般使用图论的方式进行保存，将字符串上的问题转换为图论相关问题。

AC自动机

使用范围（持续更新）

  • 多模字符串匹配，即在一个字符串集合中寻找有多少模式串在文本串中匹配。

如何构造AC自动机

使用类似Trie树的的方式进行构造。每一个状态表示为字符串集合中的一个可行前缀。转移函数为两种可能，一种是正常连接每个节点与其子节点，即表示某一模式串的更长的前缀，另一种是匹配失败的时候进行的fail转移。

由于AC自动机过水，AC自动机的构造此处仅进行简单的文字描述。在搭建Trie树的时候多记录一个fail[i]，表示第i个节点匹配失败的时候，其最长的成功匹配节点在哪里（类似KMP的失配数组），根节点的fail[]就是自己本身，根节点的子节点的fail[]还是根节点，其他的点fail[]指针指向其父节点的fail[]下的该新添加的下一个字符，如果不存在那个节点，就将该fail[]指向根节点。

如何查询AC自动机

由于AC自动机过水，AC自动机的查询此处仅进行简单的文字描述。跟着Trie树一步一步向下走，到了某一个节点的时候，从该节点的fail[]开始一直递归，直到到根节点为止，将中间递归遍历到的节点更新答案或记录，随后继续向下走，如果失配到底了，就直接走fail[]即可。

【模板】AC自动机（简单版）

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1000100;
const int MAXL = 1000100;

int n;
char input[MAXL];
int tot, ch[MAXL][30], fail[MAXL], cnt[MAXL], vis[MAXL];

int update(int &pos){return pos = (pos == 0 ? ++tot : pos);}

void insert(char *str){
    int len = strlen(str);
    int pos = 0;
    for(int i = 0; i < len; i++){
        pos = update(ch[pos][str[i] - 'a']);
    }
    cnt[pos]++;
}

void build(){
    int pos;
    queue<int> que;
    memset(fail, 0, sizeof(fail));
    for(int i = 0; i < 26; i++){
        if(ch[0][i])
            que.push(ch[0][i]);
    }
    while(!que.empty()){
        pos = que.front(); que.pop();
        for(int i = 0; i < 26; i++){
            if(ch[pos][i]){
                fail[ch[pos][i]] = ch[fail[pos]][i];
                que.push(ch[pos][i]);
            }else{
                ch[pos][i] = ch[fail[pos]][i];
            }
        }
    }
}

int query(char *str){
    memset(vis, 0, sizeof(vis));
    int len, pos, res;
    len = strlen(str);
    pos = res = 0;
    for(int i = 0; i < len; i++){
        pos = ch[pos][str[i] - 'a'];
        for(int j = pos; j && !vis[j]; j = fail[j]){
            res += cnt[j];
            vis[j] = 1;
        }
    }
    return res;
}

int main(){
    tot = 0;
    scanf("%d", &n);
    for(int i = 1; i <= n; i++){
        scanf("%s", input + 1);
        insert(input + 1);
    }
    build();
    scanf("%s", input + 1);
    int ans = query(input + 1);
    printf("%d\n", ans);
    return 0;
}

    memset(cnt, 0, sizeof(cnt));
    int len = s.size();
    int pos = 0;
    for(int i = 0; i < len; i++){
        pos = ch[pos][s[i] - 'a'];
        for(int j = pos; j; j = fail[j]){
            cnt[j]++;
        }
    }
}

int main(){
    while(scanf("%d", &n)){
        if(n == 0) break;
        tot = 0;
        memset(ch, 0, sizeof(ch));
        for(int i = 1; i <= n; i++){
            cin >> str[i];
            edpos[i] = insert(str[i]);
        }
        build();
        cin >> ques;
        query(ques);
        int ans = 0;
        for(int i = 1; i <= n; i++){
            ans = max(ans, cnt[edpos[i]]);
        }
        cout << ans << endl;
        for(int i = 1; i <= n; i++){
            if(cnt[edpos[i]] == ans)
                cout << str[i] << endl;
        }    
    }

【模板】AC自动机（二次加强版）

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200100;
const int MAXL = 200100;
const int MAXQ = 2000100;

int n, edpos[MAXN << 1];
char input[MAXQ];
int tot, ch[MAXL][30], fail[MAXL], cnt[MAXL];
int etot, frt[MAXL], nxt[MAXL], ed[MAXL];

int update(int &pos){return pos = (pos == 0 ? ++tot : pos);}

void add_edge(int u, int v){
    ed[++etot] = v;
    nxt[etot] = frt[u]; frt[u] = etot;
}

int insert(char *str){
    int len = strlen(str);
    int pos = 0;
    for(int i = 0; i < len; i++){
        pos = update(ch[pos][str[i] - 'a']);
    }
    return pos;
}

void build(){
    queue<int> que;
    int pos;
    memset(fail, 0, sizeof(fail));
    for(int i = 0; i < 26; i++){
        if(ch[0][i]) que.push(ch[0][i]);
    }
    while(!que.empty()){
        pos = que.front(); que.pop();
        for(int i = 0; i < 26; i++){
            if(ch[pos][i]){
                fail[ch[pos][i]] = ch[fail[pos]][i];
                que.push(ch[pos][i]);
            }else{
                ch[pos][i] = ch[fail[pos]][i];
            }
        }
    }
}

void query(char *str){
    memset(cnt, 0, sizeof(cnt));
    int len = strlen(str);
    int pos = 0;
    for(int i = 0; i < len; i++){
        pos = ch[pos][str[i] - 'a'];
        cnt[pos]++;
    }
}

void dfs(int u){
    int v;
    for(int i = frt[u]; i; i = nxt[i]){
        v = ed[i];
        dfs(v);
        cnt[u] += cnt[v];
    }
}

int main(){
    tot = 0;
    memset(ch, 0, sizeof(ch));
    scanf("%d", &n);
    for(int i = 1; i <= n; i++){
        scanf("%s", input + 1);
        edpos[i] = insert(input + 1);
    }
    build();
    scanf("%s", input + 1);
    query(input + 1);
    etot = 0;
    memset(frt, 0, sizeof(frt));
    for(int i = 1; i <= tot; i++) add_edge(fail[i], i);
    dfs(0);
    for(int i = 1; i <= n; i++) printf("%d\n", cnt[edpos[i]]);
    return 0;
}

多模串匹配按照此程序作为基准，建立fail树进行答案统计才可获得正确复杂度。

AC自动机总结

用于多模字符串匹配的快速算法，感觉会了AC自动机就不用KMP了？不过一般都是先学KMP的…复杂度建立Trie树为$O(\sum S_i)$，建立fail[]指针为$O(\sum S_i)$，模式串匹配为$O(S)$，建立fail树反向处理为$O(\sum S_i)$。

什么是后缀自动机

首先，后缀自动机是一种有限状态自动机，他可以识别且只识别一个字符串的后缀。但是这并不是后缀自动机强大的地方，我可以说如果把AC自动机反向插入我同样可以做到这一点。

后缀自动机真正的用处在于它可以识别每一个子串。—— lvmaooi

同时，建立一个字符串$S$的后缀自动机仅需要$O(\left|S\right|)$的复杂度。

怎么建立后缀自动机——预备知识

对于给定的一个字符串$S$，我们如何建立一个自动机来保证我们可以遍历到每一个可能的后缀呢？首先，我们会想到建立一个Trie树，比如对于一个字符串，$\underline{aabbabd}$它的一个可行的自动机（Trie）是这样子的。

  1. 有一个源点，若干终止节点。每一条边，表示在目前字符串后加上的字母，从源点到任意一个节点的任意路径可以形成一个字符串。
  2. 从源点到任意节点的任意路径形成的字符串均为原串子串。从源点到任意节点的任意路径均不能形成的字符串均不为原串子串。（简单来说，这个图可以表示，且仅可以表示出原串的所有子串）
  3. 从源点到任意终止节点的任意路径形成的字符串均为原串后缀。

  4. 从源点楚大的任意两条不同路径形成的字符串不相同。

    如果满足以上四个性质，那我们便可以用此DAG（注：DFA）处理许多事情。比如，判断某一个串是否为原串的子串（做法：从源点跑这个串，跑到NULL说明不是子串）、不同字串的个数（做法：DAG（注：DFA）上dp）等。后缀自动机可以处理的问题则多得多，因为它有更特殊的性质。—— KesdiaelKen

    绿色节点为起始状态，橙色节点是终结状态。我们可以发现直接建图的时间复杂度和空间复杂度都是$O(n^2)$级别的，并不优秀。—— Facico

原因是状态数太多了，但如果你动手自己做一下的话就会发现可以大量地拷贝复制来快速创建出Trie树，所以我们想办法如何对状态数进行压缩。

对于一个字符串$S$，它对应的后缀自动机是一个最小的确定有限状态自动机，接受且只接受$S$的后缀。对于一个DFA来说，我们可以用一个五元组<字符集、状态集、转移函数、起始状态、终结状态集>来表示它。其中比较重要的是它的状态集和转移函数。—— zjp_shadow

如果两个子串$A$和$B$的$endpos(A)=endpos(B)$的话，那么他们明显一个是另一个的后缀，假设$A$是$B$的后缀，那么他们再继续扩展都只会是一种状态，所以可以把他们合并为同一种状态。

其实后缀自动机就像是一个维护$endpos()$集合的关系的一个自动机。—— Facico

首先先定义一些接下来会使用到的概念：

  • 子串的结束位置集合函数endpos()

    对于原串$S$的一个子串$s$，$endpos(s)$表示了所有出现的结束位置的集合。

    以$S=\underline{aabbabd}$为例，$endpos(\underline{ab})=\{3,6\}$，因为$\underline{ab}$一共出现了两次，结束位置分别是$3$和$6$。同理，$endpos(\underline{a})=\{1,2,5\}$，$endpos(\underline{abba})=\{5\}$。——zjp_shadow

    $endpos(s)$有可能是空集，比如对于$S=\underline{aabbabd}$时，$endpos(\underline{ad})=\varnothing$。

以及一些会用到的性质和其证明：

  • 对于$S$的任意两个子串$s_1$和$s_2$（不妨设$\left|s_1\right|\leq\left|s_2\right|$）。则$s1$是$s2$的后缀当且仅当$endpos(s_1)\subseteq endpos(s_2)$，$s_1$不是$s_2$的后缀当且仅当$endpos(s_1)\cap endpos(s_2)=\varnothing$。由于两者关系只可能是$s_1$是$s_2$后缀或$s_1$不是$s_2$后缀两者中的一种，所以对于任意$s_1$和$s_2$，均有$endpos(s_1)\subseteq endpos(s_2)$或$endpos(s_1)\cap endpos(s_2)=\varnothing$。

    首先证明$s_1$是$s_2$的后缀时，$endpos(s_1)\subseteq endpos(s_2)$。

    由于$s_1$是$s_2$后缀，所以每次出现$s_2$的时候，$s_1$一定会伴随着出现，因此得证。

    然后证明$endpos(s_1)\subseteq endpos(s_2)$时，$s_1$是$s_2$的后缀。

    显然，$endpos(s_2)\neq\varnothing$，那么意味着每次$s_2$结束时$s_1$也会结束，且$\left|s_1\right|\leq\left|s_2\right|$，那么显然成立。（注：感性理解）

    所以两者互为充要条件。那么$s_1$不是$s_2$的后缀当且仅当$endpos(s_1)\cap endpos(s_2)=\varnothing$就是其中的推论了，后者是前者的必要条件。—— zjp_shadow

    稍微感性理解一下，还是应该很可以理解的？
