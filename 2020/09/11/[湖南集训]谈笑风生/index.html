题面

题目描述

给定一棵$n$个节点的有根数$T$，节点的编号为$1\sim n$，根节点为$1$号节点。

你需要回答$q$个询问，询问给定两个整数$p$和$k$，问有多少个有序三元组$(a,b,c)$满足：

  1. $a$、$b$和$c$为$T$中三个不同的点，且$a$为$p$号节点；
  2. $a$和$b$都是$c$的祖先；
  3. $a$和$b$在树上的距离不超过$k$。

输入

第一行含有两个正整数$n$和$q$，分别代表有根数的点数与询问的个数。

接下来$n-1$行，每行描述一条树上的边。每行含有两个整数$u$和$v$，代表在节点$u$和$v$之间有一条边。

接下来$q$行，每行描述一个操作。第$i$行含有两个整数，分别表示第$i$个询问的$p$和$k$。

输出

输出$q$行，每行对应一个询问，代表询问的答案。

样例输入1

5 3
1 2
1 3
2 4
4 5
2 2
4 1
2 3

样例输出1

3
1
3

思路与解法

对于三元组$(a,b,c)$在确定$a$的情况下分两种情况讨论：

对于所有$b$是$a$的祖先的情况，直接进行一次dfs记录每个节点的大小sz[i]和深度dep[i]，那么此时的第一部分答案为ans1 = sz[i] * min(dep[p] - 1, k)。

而对于$a$是$b$的祖先的情况，是该题选入杂题报告的原因。做法是利用记录树的欧拉序，利用欧拉序性质建立主席树维护欧拉序作为时间轴，深度作为节点的值进行计算。由于样例过于水我们使用另一组数据进行举例：

10 0
1 2
1 3
2 10
2 6
3 8
3 5
3 4
5 7
5 9

其欧拉序为1, 2, 10, 2, 6, 2, 1, 3, 8, 3, 5, 7, 5, 9, 5, 3, 4, 3, 1，用它建立主席树，比如我要算的是$p=1,k=2$的答案，那么就是主席树$1$号到$19$号节点中，深度为dep[1] + 1 = 2至min(dep[1] + k, maxdep) = 3的左右节点的sz[i]之和。以此将树上某棵子树一定深度内的节点的问题解决了。最终代码如下：

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 300100;
const int MAXM = 600100;
typedef long long ll;

ll n, q;
ll tot, frt[MAXN], nxt[MAXM], ed[MAXM];
ll maxdep, dep[MAXN], sz[MAXN];
ll cnt, rcnt, root[MAXN], tr[MAXN << 5], ch[MAXN << 5][2], idx[MAXN];

void add_edge(ll u, ll v){
    ed[++tot] = v;
    nxt[tot] = frt[u]; frt[u] = tot;
}

void dfs1(ll u, ll fa){
    ll v;
    sz[u] = 1;
    maxdep = max(maxdep, dep[u]);
    for(ll i = frt[u]; i; i = nxt[i]){
        v = ed[i];
        if(v != fa){
            dep[v] = dep[u] + 1;
            dfs1(v, u);
            sz[u] += sz[v];
        }
    }
}

void build(ll rt, ll l, ll r){
    tr[rt] = 0;
    if(l == r) return;
    ll mid = (l + r) >> 1;
    ch[rt][0] = ++cnt;
    ch[rt][1] = ++cnt;
    build(ch[rt][0], l, mid);
    build(ch[rt][1], mid + 1, r);
}

void insert(ll ort, ll rt, ll l, ll r, ll pos, ll val){
    tr[rt] = tr[ort] + val;
    if(l == r) return;
    ll mid = (l + r) >> 1;
    if(pos <= mid){
        ch[rt][0] = ++cnt;
        ch[rt][1] = ch[ort][1];
        insert(ch[ort][0], ch[rt][0], l, mid, pos, val);
    }else{
        ch[rt][0] = ch[ort][0];
        ch[rt][1] = ++cnt;
        insert(ch[ort][1], ch[rt][1], mid + 1, r, pos, val);
    }
}

ll query(ll rt, ll al, ll ar, ll l, ll r){
    if(l <= al && ar <= r) return tr[rt];
    ll mid = (al + ar) >> 1, res = 0;
    if(l <= mid) res += query(ch[rt][0], al, mid, l, r);
    if(mid + 1 <= r) res += query(ch[rt][1], mid + 1, ar, l, r);
    return res;
}

void print(ll rt, ll l, ll r){
    printf(":%d(%d)(%d, %d), lft%d, rht%d\n", rt, tr[rt], l, r, ch[rt][0], ch[rt][1]);
    if(l == r) return;
    ll mid = (l + r) >> 1;
    print(ch[rt][0], l, mid);
    print(ch[rt][1], mid + 1, r);
}

void dfs2(ll u, ll fa){
    ll v;
    idx[u] = ++rcnt;
    root[idx[u]] = ++cnt;
    insert(root[rcnt - 1], root[rcnt], 1, maxdep, dep[u], sz[u] - 1);
    for(ll i = frt[u]; i; i = nxt[i]){
        v = ed[i];
        if(v != fa){
            dfs2(v, u);
        }
    }
}

int main(){
    freopen("three.in", "r", stdin);
    freopen("three.out", "w", stdout);
    scanf("%lld %lld", &n, &q);
    for(ll u, v, i = 1; i <= n - 1; i++){
        scanf("%lld %lld", &u, &v);
        add_edge(u, v);
        add_edge(v, u);
    }
    dep[1] = 1;
    dfs1(1, 0);
//  printf(":"); for(int i = 1; i <= n; i++) printf("%d ", sz[i]); printf("\n");
//  printf(":"); for(int i = 1; i <= n; i++) printf("%d ", dep[i]); printf("\n");
    memset(root, -1, sizeof(root));
    memset(tr, -1, sizeof(tr));
    memset(ch, -1, sizeof(ch));
    root[rcnt = 0] = cnt = 1;
    build(root[0], 1, maxdep);
    dfs2(1, 0);
//  printf(":%d %d\n", idx[2], idx[2] + sz[2] - 1);
//  print(root[idx[2] + sz[2] - 1], 1, maxdep);
    for(ll p, k, i = 1; i <= q; i++){
        scanf("%lld %lld", &p, &k);
        ll ans = (sz[p] - 1) * min(k, dep[p] - 1);
//      printf(":%d\n", ans);
        ll lpos = idx[p], rpos = idx[p] + sz[p] - 1;
//      printf(":%d %d\n", lpos, rpos);
//      printf(":%d\n", query(root[rpos], 1, maxdep, dep[p] + 1, min(maxdep, dep[p] + k)));
//      printf(":%d\n", query(root[lpos - 1], 1, maxdep, dep[p] + 1, min(maxdep, dep[p] + k)));
        ans += query(root[rpos], 1, maxdep, dep[p] + 1, min(maxdep, dep[p] + k))
               - query(root[lpos - 1], 1, maxdep, dep[p] + 1, min(maxdep, dep[p] + k));
        printf("%lld\n", ans);
    }
    return 0;
}
