A. Yet Another Two Integers Problem

水，直接说结论：答案是$\left\lceil\dfrac{b-a}{10}\right\rceil$。

#include<bits/stdc++.h>

    int tskcnt = 1; scanf("%d", &tskcnt);

B. Minimum Product

尽管很水，但是考试的时候竟然卡了一会儿，因此这里给出详细解答：

题意是给定两个正整数数$a$和$b$，每次操作使a = a - 1或b = b - 1，以公共可以进行$n$次这种操作，并且操作过程中要始终满足$a\geq x$且$b\geq y$，保证$x,y\in\mathbb{Z^+}$。首先思路是考虑对于$a$和$b$没有下限的控制，此时不妨设$a\leq b$，那么每次操作如果使$a - 1$的话，最终答案减小$b$；反之对$b$操作的话答案减小$a$。所以最优解一定是使$b$不停得减小减小，一直到$b-n$最终一定是最优解。那么此时考虑上两者有下限的要求时，如果$b-n\geq y$，对于最终答案没有任何变化。如果$b-n<y$，我们会猜想最终答案一定在将$a$减少至$x$后再减小$b$与将$b$减小至$y$后再减小$a$之间。以下对这个结论进行证明：

我们可以看到是关于$i$的一个二次函数，开口向下，在$i\in\mathbb{R}$上只有最大值，因此在范围$i\in\left[y+n-b,a-x\right]$上，它的最小值一定取在区间两端即$i=y+n-b$或$i=a-x$的时候，分别是将$a$取到$x$和将$b$取到$y$的时候。

综上所述，最优情况一定在将$a$减少至$x$后再减小$b$与将$b$减小至$y$后再减小$a$之间，我们可以$O(1)$实现判断和计算。

#include<bits/stdc++.h>
using namespace std;
//const int MAXN = ;
typedef long long ll;

ll a, b, x, y, n;

int main(){
    int tskcnt = 1; scanf("%d", &tskcnt);
    while(tskcnt--){
        ll tmps;
        scanf("%lld %lld %lld %lld %lld", &a, &b, &x, &y, &n);
        ll ans1, aa, bb, xx, yy, nn;
        aa = a; bb = b; xx = x; yy = y; nn = n;
        tmps = min(nn, aa - xx);
        nn -= tmps; aa -= tmps;
        tmps = min(nn, bb - yy);
        nn -= tmps; bb -= tmps;
        ans1 = aa * bb;
        aa = b; bb = a; xx = y; yy = x; nn = n;
        tmps = min(nn, aa - xx);
        nn -= tmps; aa -= tmps;
        tmps = min(nn, bb - yy);
        nn -= tmps; bb -= tmps;
        printf("%lld\n", min(aa * bb, ans1));
    }
    return 0;
}

C. Yet Another Array Restoration

水，题目大意是让你找一个含有$x,y\in\left[1,50\right]$的长度为$n$的等差数列子序列，使数列中最大值最小。就直接做就行。

#include<bits/stdc++.h>
using namespace std;
//const int MAXN = ;
const int INF = 0x3f3f3f3f;

int n, x, y;

int main(){
    int tskcnt = 1; scanf("%d", &tskcnt);
    while(tskcnt--){
        scanf("%d %d %d", &n, &x, &y);
        int ad, apos;
        for(int d = 1; d <= 50; d++){
            if((y - x) % d == 0){
                int lcnt = x % d == 0 ? x / d - 1 : x / d;
                int lpos = x % d == 0 ? d : x % d;
                int midcnt = (y - x) / d + 1;
                if(midcnt > n) continue;
                if(n <= midcnt + lcnt){
                    ad = d;
                    apos = y - (n - 1) * d;
                    break;
                }else{
                    ad = d;
                    apos = lpos;
                    break;
                }
            }
        }
//        printf(":%d %d\n", alpos, ad);
        for(int i = 1; i <= n; i++)
            printf("%d ", apos + (i - 1) * ad);
        printf("\n");
    }
    return 0;
}

D. Decrease the Sum of Digits

水，十进制下一位一位往前，数位和减小而数本身变大当且仅当进位的时候发生，维护一个位权一位一位向前看即可。

#include<bits/stdc++.h>
using namespace std;
//const int MAXN = ;
typedef long long ll;

ll n, s;

ll calc(ll tmps){
    ll ret = 0;
    while(tmps){
        ret += tmps % 10;
        tmps /= 10;
    }
    return ret;
}

int main(){
    int tskcnt = 1; scanf("%d", &tskcnt);
    while(tskcnt--){
        scanf("%lld %lld", &n, &s);
        if(calc(n) <= s){
            printf("0\n");
            continue;
        }
        ll pwr = 1ll, ans = 0ll, tmps = n;
        while(tmps){
            if(tmps % 10 != 0){
                ans += pwr * (10 - tmps % 10);
                tmps += 10 - tmps % 10;
                if(calc(tmps) <= s){
                    printf("%lld\n", ans);
                    break;
                }
            }
            tmps /= 10; pwr *= 10;
        }
    }
    return 0;
}

E. Two Platforms

水，由题意可知物体的$y$轴没有任何意义，只要我平面放得够低就行。先按照$x$轴升序排序，然后直接从后向前维护一遍从$i$点到$n$点开始放单独一个平面最大可以接到多少物品，然后从前往后再扫一遍计算从当前点开始放第一块平面，加上之前预处理的$k$长度之后放第二块平面的最大值之和是多少，然后更新答案即可。

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 200100;

int n, k;
int x[MAXN], y[MAXN], maxi[MAXN], ans;

bool cmp(const int &x, const int &y){
    return x < y;
}

int main(){
    int tskcnt = 1; scanf("%d", &tskcnt);
    memset(maxi, 0, sizeof(maxi));
    while(tskcnt--){
        scanf("%d %d", &n, &k);
        for(int i = 1; i <= n; i++)
            scanf("%d", &x[i]);
        for(int i = 1; i <= n; i++)
            scanf("%d", &y[i]);
        sort(x + 1, x + n + 1, cmp);
        memset(maxi, 0, sizeof(maxi));
        for(int i = n; i >= 1; i--){
            int edval = x[i] + k;
            int edpos = upper_bound(x + 1, x + n + 1, edval) - x;
            maxi[i] = max(maxi[i + 1], edpos - i);
        }
        ans = 0;
        for(int i = 1; i <= n; i++){
            int tmps = upper_bound(x + 1, x + n + 1, x[i] + k) - x;
            ans = max(ans, tmps - i + maxi[tmps]);
        }
        printf("%d\n", ans);
    }
    return 0;
}

F. Subsequences of Length Two

一度以为自己无法AK了，认为这道题不可做，甚至考虑了一下建立各种自动机上计算，然后发现就是一个水得不行的dp。首先特判$t$本身是两个相同的数的情况，最优解一定是能换多少个就换多少个，然后计算答案为$\left(\begin{aligned}n\\2\end{aligned}\right)$即可。对于另一种情况我们令f[i][j][k]表示前$i$个位置，一共有$j$个t[1]，还剩下$k$次修改操作可以使用，转移方程比较复杂：

转移代价是$O(1)$，总体复杂度$O(n^2k)$可以过

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 210;

int n, ks;
char arr[MAXN], t[5];
int f[MAXN][MAXN][MAXN];

int solve(int i, int j, int k){
//    printf("query(%d, %d, %d)\n", i, j, k);
    if(i == n + 1) return 0;
    if(f[i][j][k] != -1) return f[i][j][k];
    if(arr[i] == t[1]){
        f[i][j][k] = max(f[i][j][k], solve(i + 1, j + 1, k)); //不修改t[1] 
        if(k >= 1) f[i][j][k] = max(f[i][j][k], solve(i + 1, j, k - 1) + j); //改为t[2] 
    }else if(arr[i] == t[2]){
        f[i][j][k] = max(f[i][j][k], solve(i + 1, j, k) + j); //不修改t[2]
        if(k >= 1) f[i][j][k] = max(f[i][j][k], solve(i + 1, j + 1, k - 1)); //改为t[1] 
    }else{
        f[i][j][k] = max(f[i][j][k], solve(i + 1, j, k)); //不修改
        if(k >= 1) f[i][j][k] = max(f[i][j][k], solve(i + 1, j + 1, k - 1)); //改为t[1]
        if(k >= 1) f[i][j][k] = max(f[i][j][k], solve(i + 1, j, k - 1) + j); //改为t[2]
    }
//    printf("f[%d][%d][%d] = %d\n", i, j, k, f[i][j][k]);
    return f[i][j][k];
}

int main(){
    scanf("%d %d", &n, &ks);
    scanf("%s", arr + 1);
    scanf("%s", t + 1);
    if(t[1] == t[2]){
        int cnts = 0;
        for(int i = 1; i <= n; i++){
            if(arr[i] == t[1]) cnts++;
            else if(ks){
                ks--;
                cnts++;
            }
        }
        printf("%lld\n", 1ll * cnts * (cnts - 1) / 2);
    }else{
        memset(f, -1, sizeof(f));
        printf("%d\n", solve(1, 0, ks));
    }
    return 0;
}
