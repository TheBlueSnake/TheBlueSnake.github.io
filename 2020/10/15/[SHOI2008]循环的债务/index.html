链接[SHOI2008]循环的债务

思路和解法

思路

这道题在dp题单中做到的，所以很自然想到使用dp去解决。最开始的思路钉死在了利用三者的交换差，然后使用f[i + 1000][j + 1000]表示Alice欠了Bob$i$元，Bob欠了Cynthia$j$元，可以证明一定可以用$x_1,x_2,x_3$变换为仅仅存在$i$和$j$。理论上这种方法应该是可以做的，但是貌似卡住了，以后有时间再回来做一做。【FLAG】

然后说正解，最初肯定可以想到使用f[i][j][k][p]表示Alice手中有$j$元，Bob手中有$k$元，Cynthia手中有$p$元所需要交换的最小次数。那么如果我们假设转移的是纸币$i$，它的面值是$val_i$，Alice手上最初拿了$a_i$该种纸币，Bob手上最初拿了$b_i$该种纸币，Cynthia手上最初拿了$c_i$该种纸币。我们可以获得如下的转移方程：

这时我们可以发现会有许多无用状态，所以我们可以利用三者交换纸币，但是纸币之和是不变的将冗余状态扔掉。如果我们假设三者总共有$tot$元钱，那么使用f[i][j][k]表示Alice手中有$j$元，Bob手中有$k$元，Cynthia手中有$tot-j-k$元所需要交换的最小次数。转移的纸币是纸币$i$，其他意义同上。可以得到转移方程：

此时我们可以利用该转移方程写出第一代代码。

50分代码

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1100;
const int kk[] = {-1, 100, 50, 20, 10, 5, 1};
#define INF 0x3f3f3f3f

int x1, x2, x3;
int a[10], b[10], c[10], sum[10];
int f[10][MAXN][MAXN];

int main(){
    scanf("%d %d %d", &x1, &x2, &x3);
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    memset(c, 0, sizeof(c));
    for(int i = 1; i <= 6; i++){
        scanf("%d", &a[i]);
        a[0] += a[i] * kk[i];
        sum[i] += a[i];
    }
    for(int i = 1; i <= 6; i++){
        scanf("%d", &b[i]);
        b[0] += b[i] * kk[i];
        sum[i] += b[i];
    }
    for(int i = 1; i <= 6; i++){
        scanf("%d", &c[i]);
        c[0] += c[i] * kk[i];
        sum[i] += c[i];
    }
    memset(f, 0x3f, sizeof(f));
    int tot = a[0] + b[0] + c[0];
    f[0][a[0]][b[0]] = 0;
    //a[0], b[0], c[0]是ABC三人的钱数之和
    //tot是总钱数
    //sum[i]是第i种钞票的数量 
    for(int i = 1; i <= 6; i++){
        for(int j = 0; j <= tot; j++){ //j是A有的钱数 
            for(int k = 0; k + j <= tot; k++){ //k是B有的钱数 
                for(int p = 0; p <= sum[i]; p++){ //p是A交换后有p张i类钞票 
                    for(int q = 0; q + p <= sum[i]; q++){ //q是B交换后有q张i类钞票 
                        int deltaA = j + a[i] * kk[i] - p * kk[i]; //即交换前A有多少钱 
                        int deltaB = k + b[i] * kk[i] - q * kk[i];
                        if(deltaA >= 0 && deltaB >= 0 && tot - deltaA - deltaB >= 0){
                            int numA = a[i] - p;
                            int numB = b[i] - q;
                            int numC = c[i] - (sum[i] - p - q);
                            f[i][j][k] = min(f[i][j][k], 
                            f[i - 1][deltaA][deltaB] + ((abs(numA) + abs(numB) + abs(numC)) >> 1));
                        }
                    }
                }
            }
        }
    }
    int finalA = a[0] - x1 + x3; //即最终A有多少钱 
    int finalB = b[0] - x2 + x1;
    int finalC = c[0] - x3 + x2;
    if(finalA < 0 || finalB < 0 || finalC < 0 || f[i][finalA][finalB] == INF){
        printf("impossible\n");
    }else{
        printf("%d\n", f[i][finalA][finalB]);
    }
    return 0;
}

改进与优化

该程序使用的是转移式写的程序，所以复杂度是跑满的，我们如果利用递推式进行转移的话。可以将不可能转移到的位置直接不进行第4，5层枚举转移，节省复杂度。于是我们获得了正解。

AC代码

#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1100;
const int kk[] = {-1, 100, 50, 20, 10, 5, 1};
#define INF 0x3f3f3f3f

int x1, x2, x3;
int a[10], b[10], c[10], sum[10];
int f[10][MAXN][MAXN];

int main(){
    scanf("%d %d %d", &x1, &x2, &x3);
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    memset(c, 0, sizeof(c));
    for(int i = 1; i <= 6; i++){
        scanf("%d", &a[i]);
        a[0] += a[i] * kk[i];
        sum[i] += a[i];
    }
    for(int i = 1; i <= 6; i++){
        scanf("%d", &b[i]);
        b[0] += b[i] * kk[i];
        sum[i] += b[i];
    }
    for(int i = 1; i <= 6; i++){
        scanf("%d", &c[i]);
        c[0] += c[i] * kk[i];
        sum[i] += c[i];
    }
    memset(f, 0x3f, sizeof(f));
    int tot = a[0] + b[0] + c[0];
    f[0][a[0]][b[0]] = 0;
    //a[0], b[0], c[0]是ABC三人的钱数之和
    //tot是总钱数
    //sum[i]是第i种钞票的数量 
    for(int i = 1; i <= 6; i++){
        for(int j = 0; j <= tot; j++){ //j是A有的钱数 
            for(int k = 0; k + j <= tot; k++){ //k是B有的钱数 
                if(f[i - 1][j][k] == INF) continue;
                for(int p = 0; p <= sum[i]; p++){ //p是A交换后有p张i类钞票 
                    for(int q = 0; q + p <= sum[i]; q++){ //q是B交换后有q张i类钞票 
                        int deltaA = j - a[i] * kk[i] + p * kk[i]; //即交换后A有多少钱 
                        int deltaB = k - b[i] * kk[i] + q * kk[i];
                        if(deltaA >= 0 && deltaB >= 0 && tot - deltaA - deltaB >= 0){
                            int numA = a[i] - p;
                            int numB = b[i] - q;
                            int numC = c[i] - (sum[i] - p - q);
                            f[i][deltaA][deltaB] = min(f[i][deltaA][deltaB], 
                            f[i - 1][j][k] + ((abs(numA) + abs(numB) + abs(numC)) >> 1));
                        }
                    }
                }
            }
        }
    }
    int finalA = a[0] - x1 + x3; //即最终A有多少钱 
    int finalB = b[0] - x2 + x1;
    int finalC = c[0] - x3 + x2;
    if(finalA < 0 || finalB < 0 || finalC < 0 || f[6][finalA][finalB] == INF){
        printf("impossible\n");
    }else{
        printf("%d\n", f[6][finalA][finalB]);
    }
    return 0;
}
