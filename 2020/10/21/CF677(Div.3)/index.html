A. Boring Apartments

保证每一位数字都一样的话，我们只要用x % 10取出个位数即可，然后一共需要加 $x\mod 10$ 个 $1+2+3+4=10$，最终再加上当前的位数的额外开销是 $1$ 或 $3$ 或 $6$ 或 $10$ 即可。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)
//const int MAXN = ;
typedef long long ll;
typedef pair<int, int> pii;
const int seq[] = {0, 1, 3, 6, 10};

int x;

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d", &x);
    int p = x % 10;
    int dig = 0;
    while(x){
        x = x / 10;
        dig++;
    }
    printf("%d\n", (p - 1) * 10 + seq[dig]);
}
    return 0;
}

B. Yet Another Bookshelf

需要移动的步数很明显是最左边的 $1$ 和最右边的 $1$ 中间存在的 $0$ 的个数。

#include<bits/stdc++.h>

#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)

while(_task--){
    scanf("%d", &n);
    int lpos, rpos, sum = 0;

C. Dominant Piranha

它只要求存在性，所以我们可以保证当且仅当所有数相同的情况下无解。证明如下：

充分性：如果所有数相同，很明显任何一条鱼都无法吃掉旁边的鱼，无解。

必要性：要证明无解时所有数全部相同，我们只要证明所有数不全相同时，一定有解即可。我们选取其中满足 $a_{i-1}a_{i+1}$ 的最大的 $a_i$。我们可以保证这个数一定存在，因为如果最大值两边都是和它相同的数，那么两边的数继续向外遍历一定可以在遇到边界前遇到一个比它小的数，否则的话所有数都全部相同了。

因此得证。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)
const int MAXN = 300100;
typedef long long ll;
typedef pair<int, int> pii;

int n, arr[MAXN], ans;

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d", &n);
    ans = 0;
    for(int i = 1; i <= n; i++){
        scanf("%d", &arr[i]);
    }
    int flag = 0;
    for(int i = 1; i <= n; i++){
        if(arr[i] != arr[1]){
            flag = 1;
            break;
        }
    }
    if(!flag) printf("-1\n");
    else{
        int pos = -1;
        arr[0] = arr[n + 1] = INF;
        for(int i = 1; i <= n; i++){
            if(arr[i] > ans && (arr[i - 1] < arr[i] || arr[i + 1] < arr[i])){
                ans = arr[i];
                pos = i;
            }
        }
        printf("%d\n", pos);
    }
}
    return 0;
}

D. Districts Connection

很有意思，做的时候没有第一反应做出来。上了个厕所才想出来的。

对于数列$a_i$来说，如果所有数全部相同很明显不可以满足题目要求。除此之外的话我们一定可以满足题目要求。考虑如下构造：我们选出两个不同的数 $a_i$ 和 $a_j$，那么我们首先连边 $i\to j$，对于 $k\neq i,j$，如果 $a_k\neq a_i$，那么我们连边 $k\to i$，否则我们连边 $k\to j$。如此每个点只会有一个确定的父亲节点，总共连边 $n-1$ 条边且联通，所以一定是一棵树。满足题意。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)
const int MAXN = 5100;
typedef long long ll;
typedef pair<int, int> pii;

int n, arr[MAXN];

int main(){int _task = 1; scanf("%d", &_task);
while(_task--){
    scanf("%d", &n);
    for(int i = 1; i <= n; i++){
        scanf("%d", &arr[i]);
    }
    int rtx = 1, rty;
    for(rty = 2; rty <= n; rty++){
        if(arr[rtx] != arr[rty]){
            break;
        }
    }
    if(rty == n + 1){
        printf("NO\n");
    }else{
        printf("YES\n");
        printf("%d %d\n", rtx, rty);
        for(int i = 2; i <= n; i++){
            if(i != rty){
                printf("%d %d\n", i, arr[i] == arr[rtx] ? rty : rtx);
            }
        }
    }
}
    return 0;
}

E. Two Round Dances

答案是 $\dbinom{n}{\frac n 2}\times\left[\dfrac{\left(\frac n 2\right)!}{n}\right]^2$，原因也很容易解释前面的 $\dbinom{n}{\frac n 2}$ 表示随机选出两者，后面乘以 $\left(\dfrac n 2\right)!$ 是因为一共有这么多种排列方式，但是其中每 $\dfrac n2\times2=n$ 种排列方式是重复的，所以要除掉，最终平方一下是因为上下两个队伍都可以这么变换。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)
//const int MAXN = ;
typedef long long ll;
typedef pair<int, int> pii;

ll n;

int main(){int _task = 1; //scanf("%d", &_task);
while(_task--){
    scanf("%lld", &n);
    ll ans = 1;
    for(ll i = n; i >= 1; i--){
        ans = ans * i;
    }
    printf("%lld\n", ans / n * 2 / n);
}
    return 0;
}

F. Zero Remainder Sum

明显dp。考虑使用f[i][j][p]表示在当前行，前 $i$ 个数，一共选出来 $j$ 个数，对于$k$取模后得到的和是 $p$ 的最大权值和。暴力转移 $f_{i,j,p}=\max\left(f_{i-1,j,p},f_{i-1,j-1,mod(p-a_i,k)}\right)$ 其中 $mod(i,j)$ 表示 $i$ 对于 $j$ 取模的结果。

初值赋值为-INF！！！因为这个FST了，身败名裂。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFL 0x3f3f3f3f3f3f3f3fll
#define mp(x) make_pair(x)
const int MAXN = 100;
const int MAXM = 100;
const int MAXK = 100;
typedef long long ll;
typedef pair<int, int> pii;

int n, m, s;
int arr[MAXM];
int f[MAXN][MAXM][MAXK], g[MAXK];

int main(){int _task = 1; //scanf("%d", &_task);
while(_task--){
    memset(g, 0x80, sizeof(g));
    g[0] = 0;
    memset(f, 0x80, sizeof(f));
    scanf("%d %d %d", &n, &m, &s);
    for(int i = 0; i <= m; i++) f[i][0][0] = 0;
    for(int ii = 1; ii <= n; ii++){
        for(int i = 1; i <= m; i++){
            scanf("%d", &arr[i]);
        }
        for(int i = 0; i <= m; i++){
            for(int j = 0; j <= m / 2; j++){
                for(int k = 0; k < s; k++){
                    f[i][j][k] = -INF;
                }
            }
        }
        for(int i = 0; i <= m; i++) f[i][0][0] = 0;
        for(int i = 0; i < s; i++) f[0][0][i] = g[i];
        for(int i = 1; i <= m; i++){
            for(int k = 0; k < s; k++) f[i][0][k] = f[i - 1][0][k];
            for(int j = 1; j <= m / 2; j++){
                for(int k = 0; k < s; k++){
                    f[i][j][k] = max(f[i - 1][j][k], f[i - 1][j - 1][((k - arr[i] % s) % s + s) % s] + arr[i]);
//                    printf(":f[%d][%d][%d] = %d\n", i, j, k, f[i][j][k]);
                }
            }
        }
        for(int k = 0; k < s; k++){
            int maxi = -INF;
            for(int j = 0; j <= m / 2; j++){
                maxi = max(maxi, f[m][j][k]);
//                printf(":f[%d][%d][%d] = %d\n", m, j, k, f[m][j][k]);
            }
            g[k] = maxi;
//            printf(":g[%d] = %d\n", k, g[k]);
        }
    }
    printf("%d\n", g[0]);
}
    return 0;
}

G. Reducing Delivery Cost

给你一张图，你有一个使一条边变为 $0$ 的能力，问总共 $k$ 条航线，每条航线都走最短路的最短路之和最小是多少？

最开始着迷于使用分层图，于是被卡住了。正解呢很巧妙，做 $n$ 次dijkstra，然后枚举每一条边判断如果该边变为 $0$ 的时候最短路之和最短是多少。可能的结果只有：直接走过去，不经过这条边；先走到这条边的 $x$ 然后从 $y$ 走到终点；先走到这条边的 $y$ 然后从 $x$ 走到终点三种可能性。只有后两种会更优的原因是如果要使用这条边更新答案就一定会经过这条边，经过这条边的最优解一定是最优地到达一个顶点并从另一个顶点最优地到达终点。这样做复杂度 $O(n^2\log m+mk)$，可以通过。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define INFll 0x3f3f3f3f3f3f3f3fll
#define mp(x, y) make_pair(x, y)
const int MAXN = 1100;
const int MAXM = 2100;
typedef long long ll;
typedef pair<int, int> pii;

struct edge{int u, v, w; edge(int au = 0, int av = 0, int aw = 0){u = au; v = av; w = aw;}};

int n, m, k;
int x[MAXN], y[MAXN];
int tot, frt[MAXN], nxt[MAXM]; edge ed[MAXM];
int a[MAXN], b[MAXN];
int dis[MAXN][MAXN], vis[MAXN];

void add_edge(int u, int v, int w){
    ed[++tot] = edge(u, v, w);
    nxt[tot] = frt[u]; frt[u] = tot;
}

void dijkstra(int s){
    int u, v, w;
    priority_queue<pii, vector<pii>, greater<pii> > que;
    memset(dis[s], 0x3f, sizeof(dis[s]));
    memset(vis, 0, sizeof(vis));
    que.push(mp(0, s));
    dis[s][s] = 0;
    while(!que.empty()){
        u = que.top().second; que.pop();
        if(!vis[u]){
            vis[u] = 1;
            for(int i = frt[u]; i; i = nxt[i]){
                v = ed[i].v;
                w = ed[i].w;
                if(dis[s][v] > dis[s][u] + w){
                    dis[s][v] = dis[s][u] + w;
                    que.push(mp(dis[s][v], v));
                }
            }
        }
    }
}


int main(){int _task = 1; //scanf("%d", &_task);
while(_task--){
    scanf("%d %d %d", &n, &m, &k);
    for(int w, i = 1; i <= m; i++){
        scanf("%d %d %d", &x[i], &y[i], &w);
        add_edge(x[i], y[i], w);
        add_edge(y[i], x[i], w);
    }
    for(int i = 1; i <= k; i++){
        scanf("%d %d", &a[i], &b[i]);
    }
    for(int i = 1; i <= n; i++){
        dijkstra(i);
    }
    int ans = 0, tmps;
    for(int i = 1; i <= k; i++){
        ans += dis[a[i]][b[i]];
    }
    for(int i = 1; i <= m; i++){
        tmps = 0;
        for(int j = 1; j <= k; j++){
            tmps += min(dis[a[j]][b[j]], min(dis[a[j]][x[i]] + dis[y[i]][b[j]], dis[a[j]][y[i]] + dis[x[i]][b[j]]));
        }
        ans = min(ans, tmps);
    }
    printf("%d\n", ans);
}
    return 0;
}
